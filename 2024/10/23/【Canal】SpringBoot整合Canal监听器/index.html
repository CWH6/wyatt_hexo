з®Җд»Ӣ

canal еҸҜд»Ҙз”ЁжқҘзӣ‘жҺ§ж•°жҚ®еә“ж•°жҚ®зҡ„еҸҳеҢ–пјҢд»ҺиҖҢиҺ·еҫ—ж–°еўһж•°жҚ®пјҢжҲ–иҖ…дҝ®ж”№зҡ„ж•°жҚ®гҖӮ

еүҚжҸҗ

еҮҶеӨҮеҘҪ Mysql е’Ң Navicat

дёҡеҠЎеңәжҷҜ

  • ж•°жҚ®еә“й•ңеғҸ
  • ж•°жҚ®еә“е®һж—¶еӨҮд»Ҫ
  • зҙўеј•жһ„е»әе’Ңе®һж—¶з»ҙжҠӨ(жӢҶеҲҶејӮжһ„зҙўеј•гҖҒеҖ’жҺ’зҙўеј•зӯү)
  • дёҡеҠЎ cache еҲ·ж–°
  • еёҰдёҡеҠЎйҖ»иҫ‘зҡ„еўһйҮҸж•°жҚ®еӨ„зҗҶ
  • ж•°жҚ®еә“еўһйҮҸж—Ҙеҝ—и§ЈжһҗпјҢжҸҗдҫӣеўһйҮҸж•°жҚ®и®ўйҳ…е’Ңж¶Ҳиҙ№

еҪ“еүҚзҡ„ canal ж”ҜжҢҒжәҗз«Ҝ MySQL зүҲжң¬еҢ…жӢ¬ 5.1.x , 5.5.x , 5.6.x , 5.7.x , 8.0.x

е·ҘдҪңеҺҹзҗҶ

MySQLдё»еӨҮеӨҚеҲ¶еҺҹзҗҶ

  • MySQL master е°Ҷж•°жҚ®еҸҳжӣҙеҶҷе…ҘдәҢиҝӣеҲ¶ж—Ҙеҝ—( binary log, е…¶дёӯи®°еҪ•еҸ«еҒҡдәҢиҝӣеҲ¶ж—Ҙеҝ—дәӢд»¶binary log eventsпјҢеҸҜд»ҘйҖҡиҝҮ show binlog events иҝӣиЎҢжҹҘзңӢ)
  • MySQL slave е°Ҷ master зҡ„ binary log events жӢ·иҙқеҲ°е®ғзҡ„дёӯз»§ж—Ҙеҝ—(relay log)
  • MySQL slave йҮҚж”ҫ relay log дёӯдәӢд»¶пјҢе°Ҷж•°жҚ®еҸҳжӣҙеҸҚжҳ е®ғиҮӘе·ұзҡ„ж•°жҚ®

canal е·ҘдҪңеҺҹзҗҶ

  • canal жЁЎжӢҹ MySQL slave зҡ„дәӨдә’еҚҸи®®пјҢдјӘиЈ…иҮӘе·ұдёә MySQL slave пјҢеҗ‘ MySQL master еҸ‘йҖҒdump еҚҸи®®
  • MySQL master ж”¶еҲ° dump иҜ·жұӮпјҢејҖе§ӢжҺЁйҖҒ binary log з»ҷ slave (еҚі canal )
  • canal и§Јжһҗ binary log еҜ№иұЎ(еҺҹе§Ӣдёә byte жөҒ)

// иҜҙжҳҺ: зӣ‘еҗ¬иЎЁдёӯж•°жҚ®зҡ„еҸҳеҢ–пјҲж·»еҠ гҖҒдҝ®ж”№гҖҒеҲ йҷӨпјүпјҢиҮӘеҠЁи°ғз”ЁеҜ№еә”зҡ„ж–№жі•
@ComponentG
@CanalTable("person")  // зӣ‘еҗ¬зҡ„иЎЁеҗҚ
public class PersonHandler implements EntryHandler<Person> {

    @Override
    public void insert(Person person) {
        // иЎЁдёӯж·»еҠ дәҶж•°жҚ®, зӣ‘еҗ¬е№¶жү§иЎҢдёӢйқўж“ҚдҪң
        System.out.println("иЎЁдёӯж·»еҠ дәҶж•°жҚ®");
        System.out.println(person);
    }

    @Override
    public void update(Person before, Person after) {
        // иЎЁдёӯдҝ®ж”№дәҶж•°жҚ®,зӣ‘еҗ¬е№¶жү§иЎҢдёӢйқўж“ҚдҪң
        System.out.println("иЎЁдёӯдҝ®ж”№дәҶж•°жҚ®");
        System.out.println("дҝ®ж”№еүҚпјҡ" + before);
        System.out.println("дҝ®ж”№еҗҺпјҡ" + after);
    }

    @Override
    public void delete(Person person) {
        // иЎЁдёӯеҲ йҷӨдәҶж•°жҚ®,зӣ‘еҗ¬е№¶жү§иЎҢдёӢйқўж“ҚдҪң
        System.out.println("иЎЁдёӯеҲ йҷӨдәҶж•°жҚ®");
        System.out.println(person);
    }
}
